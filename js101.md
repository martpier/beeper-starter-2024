# JavaScript 101

## 1. Variables and Basic Types

### Variable Declaration

JavaScript has three ways to declare variables:

```javascript
// const - cannot be reassigned (like a Python variable you never change)
const name = "Alice";
const age = 25;

// let - can be reassigned (most common for variables that change)
let counter = 0;
counter = 1;  // OK

// var - old way, avoid using it (has confusing scoping rules)
var oldStyle = "avoid this";
```

**Rule of thumb:** Use `const` by default, and `let` only when you need to reassign. But using `let` all the time works too.

### Basic Types

```javascript
// Numbers (no separate int/float like Python)
const integer = 42;
const decimal = 3.14;

// Strings
const single = 'Hello';
const double = "World";
const template = `Hello ${name}, you are ${age} years old`;  // Template literals!

// Booleans
const isActive = true;
const isComplete = false;

// null and undefined
const nothing = null;        // Explicitly empty
let notAssigned;             // undefined (no value assigned)
const notAssigned 2 = undefined;

// print
console.log("hello world");     // "number"
```

```javascript
if (score >= 90) {
```

### Loops

```javascript
// For loop (classic)
for (let i = 0; i < 5; i++) {
    console.log(i);  // 0, 1, 2, 3, 4
}

// While loop
let count = 0;
while (count < 3) {
    console.log(count);
    count++;
}

// For...of loop (like Python's for x in list)

    console.log(num);
}
```

## 4. Objects (Dictionaries in Python)

```javascript
// Creating objects
const person = {
    name: "Alice",
    age: 25,
    city: "Paris",
    isStudent: true
};

// Accessing properties
console.log(person.name);        // Dot notation
console.log(person["age"]);      // Bracket notation (like Python dict)

// Adding/modifying properties
person.email = "alice@example.com";
person.age = 26;

// Object destructuring (unpacking)
const { name, age } = person;
console.log(name);  // "Alice"

// Spread operator (like Python's ** unpacking)
const person2 = { ...person, age: 30, country: "France" };

// Object.keys, Object.values, Object.entries
console.log(Object.keys(person));     // ["name", "age", "city", ...]
console.log(Object.values(person));   // ["Alice", 26, "Paris", ...]
console.log(Object.entries(person));  // [["name", "Alice"], ["age", 26], ...]

// Iterating over objects
for (const key in person) {
    console.log(`${key}: ${person[key]}`);
}
```

## 5. Functions

```javascript
// Function declaration
function greet(name) {
    return `Hello, ${name}!`;
}

// Function expression
const greet2 = function(name) {
    return `Hello, ${name}!`;
};

// Arrow function
const greet3 = (name) => `Hello, ${name}!`;
const greet4 = (name, 1) => {
    return `Hello, ${name}!`;
}

console.log(power(5));     // 25 (5^2)
console.log(power(5, 3));  // 125 (5^3)

// Rest parameters (like Python's *args)
function sum(...numbers) {
    return numbers.reduce((acc, n) => acc + n, 0);
}

console.log(sum(1, 2, 3, 4));  // 10
```

### Selecting Elements

```javascript
// getElementById - select by ID (returns single element or null)
const title = document.getElementById("title");
console.log(title.textContent);  // "Hello World"

// querySelector - uses CSS selectors (returns first match)
const button = document.querySelector("#btn");
const firstPara = document.querySelector(".description");
const firstDiv = document.querySelector("div");

// querySelectorAll - returns all matches (NodeList, array-like)
const allParas = document.querySelectorAll(".description");
// Convert to array if needed: Array.from(allParas)
```

### Modifying Elements

```javascript
// Change text content
title.textContent = "Welcome!";

// Change HTML content (be careful with user input - XSS risk!)
title.innerHTML = "Welcome <strong>back</strong>!";

// Modify classes
const para = document.querySelector(".description");
para.className = "new-class";           // Replace all classes
para.classList.add("highlight");        // Add a class
para.classList.remove("description");   // Remove a class
para.classList.toggle("active");        // Toggle class on/off
para.classList.contains("highlight");   // Check if class exists

// Modify styles (camelCase for CSS properties)
title.style.color = "blue";
title.style.fontSize = "32px";
title.style.backgroundColor = "yellow";

// Modify attributes
const button = document.querySelector("#btn");
button.setAttribute("disabled", "true");
button.getAttribute("id");  // "btn"
button.removeAttribute("disabled");
```

### Creating and Inserting Elements

```javascript
// Create new element
const newDiv = document.createElement("div");
newDiv.textContent = "I'm a new div!";
newDiv.className = "dynamic-element";
newDiv.style.padding = "10px";

// Insert into DOM
const container = document.querySelector(".container");

// Append as last child
container.appendChild(newDiv);

// Insert at specific position
const firstChild = container.firstChild;
container.insertBefore(newDiv, firstChild);

// Modern methods (more flexible)
container.append(newDiv);              // Append (can add multiple)
container.prepend(newDiv);             // Add as first child
button.before(newDiv);                 // Insert before button
button.after(newDiv);                  // Insert after button

// Remove element
newDiv.remove();

// Example: Create a list dynamically
const ul = document.createElement("ul");
const items = ["Apple", "Banana", "Cherry"];

items.forEach(fruit => {
    const li = document.createElement("li");
    li.textContent = fruit;
    ul.appendChild(li);
});

document.body.appendChild(ul);
```

### Event Listeners


## 7. Asynchronous JavaScript

### Why Do We Need Async?

**Key insight:** JavaScript is **single-threaded** and runs in a browser (or Node.js) where blocking is unacceptable.

```javascript
// ‚ùå This would freeze the browser!
function sleep(ms) {
    const start = Date.now();
    while (Date.now() - start < ms) {
        // Do nothing, just wait
    }
}

console.log("Start");
sleep(3000);  // Browser freezes for 3 seconds! üò±
console.log("End");
```

**Problems with blocking:**
- User can't scroll, click, or interact
- Browser UI freezes
- Multiple network requests would take forever

**Solution:** Asynchronous code! Let JavaScript do other work while waiting.

---

### The Evolution: Callbacks ‚Üí Promises ‚Üí Async/Await

### 1. Callbacks (The Old Way)

A callback is a function passed as an argument to be called later.

```javascript
// Simulating an API call
function fetchUser(userId, callback) {
    console.log("Fetching user...");
    setTimeout(() => {
        const user = { id: userId, name: "Alice" };
        callback(user);
    }, 1000);
}

fetchUser(1, (user) => {
    console.log("Got user:", user);
});

console.log("This runs immediately!");
// Output:
// Fetching user...
// This runs immediately!
// Got user: { id: 1, name: 'Alice' }  (after 1 second)
```

**Problem: Callback Hell** üî•

When you need to chain multiple async operations:

```javascript
fetchUser(1, (user) => {
    fetchPosts(user.id, (posts) => {
        fetchComments(posts[0].id, (comments) => {
            fetchLikes(comments[0].id, (likes) => {
                // This is hard to read and maintain!
                console.log(likes);
            });
        });
    });
});
```

---

### 2. Promises (Better!)

A Promise represents a value that will be available in the future.

```javascript
function fetchUser(id) {
    return fetch("http://faviji/user/1)
}

fetchUser(1) // returns a promise
    .then((user) => {
        console.log("Got user:", user);
        return fetchPosts(user.id);  // Return another promise
    })
    .then((posts) => {
        console.log("Got posts:", posts);
        return fetchComments(posts[0].id);
    })
    .then((comments) => {
        console.log("Got comments:", comments);
    })
    .catch((error) => {
        console.error("Error:", error);
    });
```

**Promise states:**
- **Pending:** Initial state, waiting
- **Fulfilled:** Operation completed successfully (resolved)
- **Rejected:** Operation failed

---

### 3. Async/Await (The Modern Way)

Makes async code look synchronous and easier to read.

```javascript
// An async function always returns a Promise
async function getUserData(userId) {
    try {
        console.log("Fetching user...");
        const user = await fetchUser(userId);      // Wait for promise
        console.log("Got user:", user);

        const posts = await fetchPosts(user.id);   // Wait for next promise
        console.log("Got posts:", posts);

        const comments = await fetchComments(posts[0].id);
        console.log("Got comments:", comments);

        return comments;  // Automatically wrapped in Promise
    } catch (error) {
        console.error("Error:", error);
    }
}

// Call it
getUserData(1);

// Or with .then() since it returns a Promise
getUserData(1).then((comments) => {
    console.log("All done!", comments);
});
```

**Key points:**
- `async` keyword makes a function return a Promise
- `await` pauses execution until Promise resolves
- Only use `await` inside `async` functions
- Use `try/catch` for error handling

### Real Example: Fetch API

The `fetch()` API is how you make HTTP requests in modern JavaScript.

```javascript
// Basic fetch
fetch("https://jsonplaceholder.typicode.com/users/1")
    .then(response => response.json())  // Parse JSON
    .then(data => {
        console.log(data);
    })
    .catch(error => {
        console.error("Error:", error);
    });

// With async/await (cleaner!)
async function getUser(id) {
    try {
        const response = await fetch(`https://jsonplaceholder.typicode.com/users/${id}`);

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const user = await response.json();
        console.log(user);
        return user;
    } catch (error) {
        console.error("Failed to fetch user:", error);
    }
}

getUser(1);
```

### Parallel Async Operations

```javascript
// Sequential (slow - waits for each)
async function fetchAllSequential() {
    const user1 = await fetchUser(1);
    const user2 = await fetchUser(2);
    const user3 = await fetchUser(3);
    return [user1, user2, user3];
}

// Parallel (fast - all at once!)
async function fetchAllParallel() {
    const promises = [
        fetchUser(1),
        fetchUser(2),
        fetchUser(3)
    ];

    const users = await Promise.all(promises);
    return users;
}

// Promise.all - wait for all
const results = await Promise.all([promise1, promise2, promise3]);

// Promise.race - wait for first to complete
const fastest = await Promise.race([promise1, promise2, promise3]);
```

### Practical Exercise: Dynamic User List

```html
<!DOCTYPE html>
<html>
<head>
    <title>User List</title>
    <style>
        .user-card {
            border: 1px solid #ccc;
            padding: 10px;
            margin: 10px 0;
            border-radius: 5px;
        }
        .loading {
            color: gray;
            font-style: italic;
        }
    </style>
</head>
<body>
    <h1>User List</h1>
    <button id="loadUsers">Load Users</button>
    <div id="userList"></div>

    <script>
        const button = document.getElementById("loadUsers");
        const userList = document.getElementById("userList");

        async function loadUsers() {
            // Show loading message
            userList.innerHTML = '<p class="loading">Loading...</p>';

            try {
                const response = await fetch("https://jsonplaceholder.typicode.com/users");
                const users = await response.json();

                // Clear loading message
                userList.innerHTML = "";

                // Create user cards
                users.forEach(user => {
                    const card = document.createElement("div");
                    card.className = "user-card";
                    card.innerHTML = `
                        <h3>${user.name}</h3>
                        <p>Email: ${user.email}</p>
                        <p>City: ${user.address.city}</p>
                    `;
                    userList.appendChild(card);
                });
            } catch (error) {
                userList.innerHTML = `<p style="color: red;">Error: ${error.message}</p>`;
            }
        }

        button.addEventListener("click", loadUsers);
    </script>
</body>
</html>
```
